#include<iostream>
using namespace std;
const int MAX=100;

int Swap(int a, int b)
{
    int swap;
    swap=a;
    a=b;
    b=swap;
}

void QSort(int arr[], int l, int r)
{

    int i = l, j = r, x = arr[(i + j) / 2];
    while (i <= j)
    {
        while (arr[i] < x)
            i++;
        while (arr[j] > x)
            j--;
        if (i <= j)
        {
            Swap(arr[i],arr[j]);
            i++;
            j--;
        }
    }

    if (j > l) QSort(arr, l, j);
    if (i < r) QSort(arr, i+1, r);
}

int main()
{
    int n,l,r, arr[MAX];
    cout<<"Elements in array= ";
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    l=0;
    r=n-1;
    QSort(arr,l,r);

    for(int i=0;i<n;i++)
    {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    return 0;
}
