квадратна целочислена таблица

#include<iostream>

using namespace std;

int main()
{

    /*
    1 2 3 Е n-1 n
    2 3 4 Е n n+1
    3 4 5 Е n+1 n+2
    Е Е Е
    n-1 n n+1 Е 2n-3 2n-2
    n n+1 n+2 Е 2n-2 2n-1

     */

    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        for(int j=i;j<=n+i-1;j++)
        {
            cout<<j<<" ";
        }
        cout<<endl;
    }
    return 0;
}
