Сортиране чрез пряка селекция:

1 Намира най-малкия елемент в списъка като сравнява първият елемент с всички останали
2 Разменя го с елемента на първа позиция
3 Повтаря горните две стъпки за всеки следващ елемент


#include<iostream>
using namespace std;
const int MAX=100;

int main()
{
    int n, swap, arr[MAX];
    cout<<"Elements in array= ";
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }

    for(int i=0;i<n;i++)
    {
        int min=i;
        for(int j=i;j<n;j++)
        {
            if(arr[j]<arr[min])
                min=j;
        }
        swap=arr[min];
        arr[min]=arr[i];
        arr[i]=swap;
    }
    for(int i=0;i<n;i++)
    {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    return 0;
}
