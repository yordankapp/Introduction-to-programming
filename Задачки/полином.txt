Rl(x) = Pn(x) * Qm(x);

#include<iostream>
using namespace std;
const int MAX=100;


void Mult(int a[], int b[], int st_a, int st_b, int c[], int& st_c)
{
    st_c=st_a+st_b;
    cout<<st_c<<endl;
    for(int i=0;i<=st_c;i++)
    {
        c[i]=0;
        for(int j=0;j<=st_a;j++)
        {
            for(int f=0;f<=st_b;f++)
            {
               if(j+f==i)
                 c[i]=c[i]+a[j]*b[i-j];
            }
        }
    }
    for(int i=0;i<=st_c;i++)
    {
        cout<<c[i]<<" ";
    }
}
