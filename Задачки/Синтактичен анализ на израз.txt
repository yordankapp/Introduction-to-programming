#include<iostream>
#include<string.h>

using namespace std;

void Sum(const char st[], int& i, int& err);
void Term(const char st[], int& i, int& err);
void Number(const char st[], int& i, int& err);

void Sum(const char st[], int& i, int& err)
{
    if (i > strlen(st)-1)
    {
        err = 1;
        return;
    }
    else
    {
        Term(st, i, err);
        i++;
        if (!err && i<strlen(st))
        {
            if (st[i] == ')' )
                return;
            if (st[i++] == '+')
            {
                Term(st, i, err);
                i++;
            }
            else
                err = 1;
        }
    }
}
void Term(const char st[], int& i, int& err)
{
    if (i > strlen(st)-1)
    {
        err = 1;
        return;
    }
    if (st[i] == '(' )
    {
        i++;
        Sum(st, i, err);
        if (!err && i < strlen(st))
            if (!(st[i]==')' ))
                err = 1;
    }
    else
    {
        Number(st, i, err);
    }
}
void Number(const char st[], int& i, int& err)
{
    if (i>strlen(st))
    {
        err = 1;
        return;
    }
    else
    {
        while (isdigit(st[i]))
        i++;
        i--;
    }
}
int main()
{
    char expr[200];
    int index = 0, error = 0;
    cout << "Въведете низ:";
    cin.getline(expr, 200);
    Sum(expr, index, error);
    if (!error && (index == strlen(expr)))
        cout << "Правилен запис на сума \n";
    else
        cout << "Неправилен запис на сума \n";
    return 0;
}
