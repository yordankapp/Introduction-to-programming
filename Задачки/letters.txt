#include<iostream>

using namespace std;

int main()
{
    /* програма, която въвежда главна латинска буква и извежда
       съответната малка буква
     */

    char letter;
    cin>>letter;
    letter+='a'-'A';
    cout<<letter<<endl;
    return 0;
}

#include<iostream>

using namespace std;

int main()
{
    /* програма, която въвежда малка латинска буква и извежда
       съответната главна буква
     */

    char letter;
    cin>>letter;
    letter-='a'-'A';
    cout<<letter<<endl;
    return 0;
}
