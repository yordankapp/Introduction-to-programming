#include <iostream>
using namespace std;
const int MAX=100;

void  transpose(int n,int m,int **arr)
{
    int **arrtranspose;

    arrtranspose = new int *[m]; //creates a new array of pointers to int objects
    for (int i = 0; i < m; i++)
        arrtranspose[i] = new int[n];

    for(int i=0; i<m; i++)
        for(int j=0; j<n; j++)
            arrtranspose[i][j] = arr[j][i];

    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++)
        {
            cout<<arrtranspose[i][j]<<" ";
        }
        cout<<endl;
    }

    for(int i=0;i<m;i++)
    {
            delete[] arrtranspose[i];
    }
    delete []arrtranspose;
}

int main()
{
    int n,m,**arr;
    cin>>n>>m;

    //make matrix
    arr = new int *[n];
    for(int i=0;i<n;i++)
    {
       arr[i]=new int[m];
    }

    //fill matrix
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
            cin>>arr[i][j];
    }

    transpose(n,m,arr);

    for(int i=0;i<n;i++)
    {
        delete[] arr[i];
    }
    delete []arr;
    return 0;
}
